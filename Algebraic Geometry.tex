\section{Varieties}
\begin{definition}
    Let $K$ be an algebraic closed field. The \hdef{affine n-space} ${\bb{A}}_{K}^{n}$ is a set $\{({a}_{1},\dots,{a}_{n})\mid{a}_{i}\in K\}$. An element $P\in{\bb{A}}_{K}^{n}$ is called a \hdef{point}, if $P=({a}_{1},\dots,{a}_{n})$, each ${a}_{i}$ is called a \hdef{coordinate} of $P$.
\end{definition}
\begin{remark}
    We will write ${\bb{A}}^{n}$ for ${\bb{A}}_{K}^{n}$.
\end{remark}
\par
Let $A=K[{x}_{1},\dots,{x}_{n}]$ be a polynomial ring, $A$ is a function such that for all $f\in A$ and $P=({a}_{1},\dots,{a}_{n})\in{\bb{A}}^{n}$, $f(P)=f({a}_{1},\dots,{a}_{n})$, which substitues ${x}_{i}$ by ${a}_{i}$. 
\begin{definition}
    Let $K$ be an algebraic closed field and $A=K[{x}_{1},\dots,{x}_{n}]$. Let $T\sub A$, the \hdef{zero set} of $T$ is the set $Z(T)=\{P\in{\bb{A}}^{n}\mid f\in A\ \text{and}\ f(P)=0\}$.
\end{definition}
